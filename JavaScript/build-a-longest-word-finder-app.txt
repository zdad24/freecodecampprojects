function findLongestWordLength(sentence) {
  let count = 0;
  let maxcount = 0;
  for (let char of sentence) {
    if (char === " ") {
      maxcount = Math.max(count, maxcount);
      count = 0;
      
    }
    else {
      count++;
    }
  }
    maxcount = Math.max(count, maxcount);
    return maxcount;
        
  }
  
console.log(findLongestWordLength("The quick brown fox jumped over the lazy dog"));
